# ===== Cell 6b (CONDITIONAL): Force-unzip remaining DICOM zips =====
# Run this if Cell 7 FAILS with "DICOM dir missing" errors.
# Yesterday's regen wrote SUV NIfTIs for all 50 cases, but the DICOM dirs for the
# 17 zip-only cases did not persist on Drive (temp-dir during yesterday's regen,
# or Drive metadata sync). The driver needs DICOM access for per-case acquisition
# parameter extraction. This cell extracts the zips so the driver can find them.

import os, zipfile
import pandas as pd
from pathlib import Path

DATA_ROOT = '/content/drive/MyDrive/P79 Data'
OUT_DIR = f'{DATA_ROOT}/phase2'
PRE_CSV = f'{OUT_DIR}/phase2_preflight_autopet_iii.csv'

pre = pd.read_csv(PRE_CSV)
todo = pre[pre['needs_dicom_unzip']]
print(f'{len(todo)} zips to extract')
print()

n_done = 0
n_failed = 0
for i, (_, r) in enumerate(todo.iterrows(), 1):
    series_uid = r['series_uid']
    dicom_zip = Path(r['dicom_zip'])
    dicom_dir = Path(r['dicom_dir'])

    if dicom_dir.exists() and any(dicom_dir.iterdir()):
        print(f'[{i}/{len(todo)}] {series_uid[:40]}... -- already populated; skip')
        continue

    if not dicom_zip.exists():
        print(f'[{i}/{len(todo)}] {series_uid[:40]}... -- zip path missing; skip')
        n_failed += 1
        continue

    try:
        zip_size_mb = dicom_zip.stat().st_size / (1024 * 1024)
        dicom_dir.mkdir(parents=True, exist_ok=True)
        with zipfile.ZipFile(dicom_zip, 'r') as zf:
            zf.extractall(dicom_dir)
        # Some zips wrap content in a nested subdirectory; flatten if so.
        inside = list(dicom_dir.iterdir())
        if len(inside) == 1 and inside[0].is_dir():
            nested = inside[0]
            for f in list(nested.iterdir()):
                f.rename(dicom_dir / f.name)
            nested.rmdir()
        n_done += 1
        print(f'[{i}/{len(todo)}] {series_uid[:40]}... -- extracted ({zip_size_mb:.1f} MB)')
    except Exception as e:
        print(f'[{i}/{len(todo)}] {series_uid[:40]}... -- FAILED: {e}')
        n_failed += 1

print()
print(f'Done: {n_done} extracted, {n_failed} failed')

import shutil, os, hashlib

DATA_ROOT = '/content/drive/MyDrive/P79 Data'
OUT_DIR = f'{DATA_ROOT}/phase2'
OUT_PREFIX_PRIMARY = f'{OUT_DIR}/phase2_autopet_iii_primary'

STAGE_SMALL = f'{OUT_DIR}/osf_stage'
STAGE_LARGE = f'{OUT_DIR}/drive_only'
os.makedirs(STAGE_SMALL, exist_ok=True)
os.makedirs(STAGE_LARGE, exist_ok=True)

def _sha256(p):
    h = hashlib.sha256()
    with open(p, 'rb') as fh:
        for chunk in iter(lambda: fh.read(1 << 20), b''):
            h.update(chunk)
    return h.hexdigest()

SMALL = [
    (f'{OUT_PREFIX_PRIMARY}_wcv.parquet', 'phase2_autopet_iii_primary_wcv.parquet'),
]
LARGE = [
    (f'{OUT_PREFIX_PRIMARY}_replicates.parquet', 'phase2_autopet_iii_primary_replicates.parquet'),
]

print('--- small artefacts (will pull back to local for OSF) ---')
for src, name in SMALL:
    if not os.path.exists(src):
        print(f'  MISSING: {src}'); continue
    dst = os.path.join(STAGE_SMALL, name)
    shutil.copy(src, dst)
    print(f'  {name}  {os.path.getsize(dst):>10,} bytes  SHA-256 {_sha256(dst)}')

print()
print('--- large artefacts (stay on Drive) ---')
for src, name in LARGE:
    if not os.path.exists(src):
        print(f'  MISSING: {src}'); continue
    dst = os.path.join(STAGE_LARGE, name)
    shutil.copy(src, dst)
    print(f'  {name}  {os.path.getsize(dst):>10,} bytes  SHA-256 {_sha256(dst)}')
