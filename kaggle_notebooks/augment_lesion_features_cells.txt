# ===== Cell A1: Stage HECKTOR zip locally (idempotent; SHA-checks if already staged) =====
import os, shutil, hashlib
from pathlib import Path

DRIVE_ZIP_HEK = Path('/content/drive/MyDrive/HECKTOR 2025/hecktor_2025_training_defaced.zip')
LOCAL_ZIP_HEK = Path('/content/hecktor_2025_training_defaced.zip')
EXPECTED_HEK_PREFIX = '1abcf1d9'

def _file_sha256(p, chunk=2**20):
    h = hashlib.sha256()
    with open(p, 'rb') as f:
        for buf in iter(lambda: f.read(chunk), b''):
            h.update(buf)
    return h.hexdigest()

if LOCAL_ZIP_HEK.exists() and _file_sha256(LOCAL_ZIP_HEK).startswith(EXPECTED_HEK_PREFIX):
    print(f'Local HECKTOR zip already staged and SHA-verified.')
else:
    print(f'Copying {DRIVE_ZIP_HEK} -> {LOCAL_ZIP_HEK} (~10-15 min)...')
    shutil.copy2(DRIVE_ZIP_HEK, LOCAL_ZIP_HEK)
    sha = _file_sha256(LOCAL_ZIP_HEK)
    print(f'Local zip SHA-256: {sha}')
    assert sha.startswith(EXPECTED_HEK_PREFIX), 'HECKTOR zip integrity check failed'
ZIP_PATH = str(LOCAL_ZIP_HEK)

import zipfile, time
import pandas as pd
import numpy as np
import SimpleITK as sitk
from pathlib import Path

from src.preprocess.hecktor_lesions import extract_hecktor_lesions
from src.preprocess.hecktor_vendor import lookup_centre

DATA_ROOT = 'HECKTOR 2025 Training Data Defaced ALL/'
EHR_MEMBER = f'{DATA_ROOT}HECKTOR_2025_Training_EHR_with_Data_Dictionary.xlsx'

OUT_DIR_HEK = '/content/drive/MyDrive/P79 Data/hecktor'
os.makedirs(OUT_DIR_HEK, exist_ok=True)

with zipfile.ZipFile(ZIP_PATH) as zf:
    with zf.open(EHR_MEMBER) as fp:
        ehr = pd.read_excel(fp, sheet_name='Data')
cohort = ehr[(ehr['Task 1'] == 1) | (ehr['Task 2'] == 1)].copy()

def _read_nifti_from_zip(zf, member):
    with zf.open(member) as fp:
        data = fp.read()
    tmp = f'/tmp/_hek_{hash(member) % 10**9}.nii.gz'
    with open(tmp, 'wb') as f:
        f.write(data)
    img = sitk.ReadImage(tmp)
    os.remove(tmp)
    return img

def _resample_seg_to_pt(seg_img, pt_img):
    rs = sitk.ResampleImageFilter()
    rs.SetReferenceImage(pt_img)
    rs.SetInterpolator(sitk.sitkNearestNeighbor)
    rs.SetDefaultPixelValue(0)
    return rs.Execute(seg_img)

t0 = time.time()
all_rows = []
n_done = 0
n_failed = 0
with zipfile.ZipFile(ZIP_PATH) as zf:
    for i, (_, r) in enumerate(cohort.iterrows(), 1):
        pid = r['PatientID']
        try:
            pt = _read_nifti_from_zip(zf, f'{DATA_ROOT}{pid}/{pid}__PT.nii.gz')
            seg = _read_nifti_from_zip(zf, f'{DATA_ROOT}{pid}/{pid}.nii.gz')
            seg_on_pt = _resample_seg_to_pt(seg, pt)
            suv = sitk.GetArrayFromImage(pt).astype(np.float32)
            seg_arr = sitk.GetArrayFromImage(seg_on_pt).astype(np.uint8)
            spacing_xyz = pt.GetSpacing()
            spacing_zyx = (float(spacing_xyz[2]), float(spacing_xyz[1]), float(spacing_xyz[0]))

            centre_id = r.get('CenterID')
            cname, vendor = lookup_centre(centre_id)
            meta = {
                'task1_patient': bool(r.get('Task 1') == 1),
                'task2_patient': bool(r.get('Task 2') == 1),
                'Relapse':  r.get('Relapse'),  'RFS': r.get('RFS'),
                'T-stage':  r.get('T-stage'),  'N-stage': r.get('N-stage'),
                'M-stage':  r.get('M-stage'),  'HPV Status': r.get('HPV Status'),
            }
            rows = extract_hecktor_lesions(
                suv=suv, seg_multi=seg_arr, voxel_spacing_mm=spacing_zyx,
                case_id=pid,
                centre_id=int(centre_id) if pd.notna(centre_id) else None,
                centre_name=cname, vendor=vendor, patient_meta=meta,
            )
            all_rows.extend([row.__dict__ for row in rows])
            n_done += 1
            if n_done % 100 == 0:
                rate = n_done / (time.time() - t0)
                print(f'  [{i}/{len(cohort)}] {pid}  rate={rate:.2f} pat/s')
        except KeyError as e:
            # Task2-only patients have no SEG -- expected, not a real failure
            n_failed += 1
        except Exception as e:
            print(f'  [{i}/{len(cohort)}] {pid} FAILED: {e}')
            n_failed += 1

df = pd.DataFrame(all_rows)
df.to_parquet(OUT_PARQUET_HEK, index=False)
print(f'\nHECKTOR v2 extraction: {n_done} succeeded, {n_failed} skipped/failed in {(time.time()-t0)/60:.1f} min')
print(f'Total lesions: {len(df)}; columns: {sorted(df.columns)}')
print(f'  surface_area_cm2 min/median/max: {df["surface_area_cm2"].min():.2f} / '
      f'{df["surface_area_cm2"].median():.2f} / {df["surface_area_cm2"].max():.2f}')
print(f'  sphericity      min/median/max: {df["sphericity"].min():.3f} / '
      f'{df["sphericity"].median():.3f} / {df["sphericity"].max():.3f}')

# ===== Cell C2: Build patient -> SEG path lookup from zip namelist =====
import zipfile

with zipfile.ZipFile(ZIP_PATH_FDAT, 'r') as zf:
    names = zf.namelist()

# FDAT layout: fdg-pet-ct-lesions/{patient_id}/{study_date}/SEG.nii.gz
# Build dict: patient_id -> list of (study_date, seg_member)
from collections import defaultdict
fdat_segs = defaultdict(list)
for n in names:
    if not n.endswith('SEG.nii.gz'):
        continue
    parts = n.split('/')
    petct_idx = next((i for i, p in enumerate(parts) if p.startswith('PETCT_')), None)
    if petct_idx is None or petct_idx + 1 >= len(parts):
        continue
    pid = parts[petct_idx]
    sdate = parts[petct_idx + 1]
    fdat_segs[pid].append((sdate, n))

print(f'FDAT zip: {len(fdat_segs)} patients with SEG files; '
      f'{sum(len(v) for v in fdat_segs.values())} total SEG entries')


# ===== Cell C3: Augment AutoPET-I parquet =====
import time, os, zipfile, tempfile
import pandas as pd
import numpy as np
import nibabel as nib

from src.features.augment_lesion_features import augment_case

AUTOPET_I_LESIONS_OUT = '/content/drive/MyDrive/P79 Data/autopet_i/lesion_features_v3.parquet'

# Resolve input parquet from any of these (priority order):
#   1. /content/drive/MyDrive/P79 Data/autopet_i/lesion_features_v2.parquet  (Drive)
#   2. /content/p79/data/interim/lesion_tables/autopet_i_lesions.parquet  (rebuilt zip 2026-04-30+)
#   3. /content/autopet_i_lesions.parquet  (dragged in via Files panel)
CANDIDATE_PARQUETS_I = [
    '/content/drive/MyDrive/P79 Data/autopet_i/lesion_features_v2.parquet',
    '/content/p79/data/interim/lesion_tables/autopet_i_lesions.parquet',
    '/content/autopet_i_lesions.parquet',
]
AUTOPET_I_LESIONS_IN = next((p for p in CANDIDATE_PARQUETS_I if os.path.exists(p)), None)
assert AUTOPET_I_LESIONS_IN, (
    'AutoPET-I parquet not found at any expected location.\n'
    'Drag autopet_i_lesions.parquet from your local Mac (data/interim/lesion_tables/) '
    'into the Colab Files panel, then re-run this cell.'
)
print(f'Using parquet: {AUTOPET_I_LESIONS_IN}')

df = pd.read_parquet(AUTOPET_I_LESIONS_IN)
print(f'{len(df)} lesions across {df["case_id"].nunique()} patients')

t0 = time.time()
augmented_chunks = []
unmatched_total = 0
n_done = 0
n_failed = 0

with zipfile.ZipFile(ZIP_PATH_FDAT, 'r') as zf:
    for case_id, case_rows in df.groupby('case_id'):
        seg_entries = fdat_segs.get(case_id, [])
        if not seg_entries:
            print(f'  NO SEG in zip for {case_id}; passing through with NaN')
            miss = case_rows.copy()
            miss['surface_area_cm2'] = float('nan')
            miss['sphericity'] = float('nan')
            augmented_chunks.append(miss)
            n_failed += 1
            continue

        # If patient has multiple study_dates, pick the one whose SEG produces
        # components matching the parquet's centroid+n_voxels signature.
        best_match_chunk = None
        best_match_score = -1
        for sdate, seg_member in seg_entries:
            try:
                with zf.open(seg_member) as fp:
                    seg_bytes = fp.read()
                # Write to /tmp briefly for nibabel
                with tempfile.NamedTemporaryFile(suffix='.nii.gz', delete=False) as tmp:
                    tmp.write(seg_bytes)
                    tmp_path = tmp.name
                seg_img = nib.load(tmp_path)
                arr = np.asarray(seg_img.get_fdata())
                # NO transpose: the original AutoPET-I extraction (process_autopet_i.ipynb
                # Step 4) used nibabel.get_fdata() without transposing, so the parquet's
                # centroid_0/1/2 columns are in nibabel's native (I, J, K) order. We must
                # match that here or centroid sanity-checks fail (~99% mismatches).
                seg_arr = arr.astype(np.int32)
                spacing = seg_img.header.get_zooms()[:3]  # (sx, sy, sz) per nibabel
                spacing_ijk = (float(spacing[0]), float(spacing[1]), float(spacing[2]))
                os.remove(tmp_path)

                out_df, result = augment_case(
                    case_id=case_id, case_lesions=case_rows,
                    seg=seg_arr, voxel_spacing_zyx=spacing_ijk,
                )
                if result.n_lesions_matched > best_match_score:
                    best_match_score = result.n_lesions_matched
                    best_match_chunk = (out_df, result)
            except Exception:
                continue
        if best_match_chunk is None:
            print(f'  {case_id}: no candidate SEG produced any matches')
            miss = case_rows.copy()
            miss['surface_area_cm2'] = float('nan')
            miss['sphericity'] = float('nan')
            augmented_chunks.append(miss)
            n_failed += 1
            continue
        out_df, result = best_match_chunk
        unmatched_total += result.n_lesions_unmatched
        augmented_chunks.append(out_df)
        n_done += 1
        if n_done % 50 == 0:
            rate = n_done / (time.time() - t0)
            print(f'  [{n_done}/{df["case_id"].nunique()}]  rate={rate:.2f} pat/s  unmatched={unmatched_total}')

augmented = pd.concat(augmented_chunks, ignore_index=True)
augmented.to_parquet(AUTOPET_I_LESIONS_OUT, index=False)

n_with_sa = int(augmented['surface_area_cm2'].notna().sum())
print(f'\nAutoPET-I augmentation:')
print(f'  patients done={n_done}, patients failed={n_failed}, unmatched lesions={unmatched_total}')
print(f'  rows with SA+sphericity: {n_with_sa}/{len(augmented)} ({n_with_sa/len(augmented)*100:.1f}%)')
print(f'Wrote {AUTOPET_I_LESIONS_OUT}')

def _sha256(p):
    h = hashlib.sha256()
    with open(p, 'rb') as fh:
        for chunk in iter(lambda: fh.read(1<<20), b''):
            h.update(chunk)
    return h.hexdigest()
