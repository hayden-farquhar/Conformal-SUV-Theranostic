# ===== Cell 4: Stage HECKTOR zip locally =====
# Cribbed from P80 §1.5: Drive FUSE I/O is 5-10x slower than local SSD; pre-stage once
# saves ~15-30 min on a 30-60 min extraction job. Idempotent + SHA-verified.

import os, shutil, hashlib
from pathlib import Path

DRIVE_ZIP = Path('/content/drive/MyDrive/HECKTOR 2025/hecktor_2025_training_defaced.zip')
LOCAL_ZIP = Path('/content/hecktor_2025_training_defaced.zip')
EXPECTED_SHA = '1abcf1d9'  # P80 receipt; full hash logged in P80 PROGRESS.md

assert DRIVE_ZIP.exists(), f'HECKTOR zip not on Drive at {DRIVE_ZIP}'

def _file_sha256(p, chunk=2**20):
    h = hashlib.sha256()
    with open(p, 'rb') as f:
        for buf in iter(lambda: f.read(chunk), b''):
            h.update(buf)
    return h.hexdigest()

def stage_zip_locally():
    if LOCAL_ZIP.exists():
        sha = _file_sha256(LOCAL_ZIP)
        if sha.startswith(EXPECTED_SHA):
            print(f'Local zip exists and matches P80 receipt prefix {EXPECTED_SHA}; reusing')
            return LOCAL_ZIP
        print(f'Local zip exists but SHA {sha[:16]}... does NOT match {EXPECTED_SHA}; re-staging')
        LOCAL_ZIP.unlink()
    # Disk space check: HECKTOR zip is ~48 GB, need ~53 GB free
    free_gb = shutil.disk_usage('/content').free / (1024**3)
    print(f'Free space on /content: {free_gb:.1f} GB')
    assert free_gb > 53, 'Need >53 GB free on /content for HECKTOR zip stage'
    print(f'Copying {DRIVE_ZIP} -> {LOCAL_ZIP} (~10-15 min)...')
    shutil.copy2(DRIVE_ZIP, LOCAL_ZIP)
    sha = _file_sha256(LOCAL_ZIP)
    print(f'Local zip SHA-256 (full): {sha}')
    print(f'P80 receipt prefix: {EXPECTED_SHA}')
    if not sha.startswith(EXPECTED_SHA):
        print(f'WARNING: SHA does not match P80 receipt prefix; verify before proceeding')
    return LOCAL_ZIP

ZIP_PATH = stage_zip_locally()
print(f'Using zip at: {ZIP_PATH}')

# ===== Cell 6: Smoke test (3 patients, full pipeline) =====
import time
import numpy as np
import SimpleITK as sitk
import zipfile, io

from src.preprocess.hecktor_lesions import extract_hecktor_lesions
from src.preprocess.hecktor_vendor import lookup_centre


def read_nifti_from_zip(zf, member_name):
    """Read a NIfTI from zip without extracting to disk."""
    with zf.open(member_name) as fp:
        data = fp.read()
    # SimpleITK needs a file path; write to /tmp briefly
    tmp = f'/tmp/_hecktor_{hash(member_name) % 10**9}.nii.gz'
    with open(tmp, 'wb') as f:
        f.write(data)
    img = sitk.ReadImage(tmp)
    os.remove(tmp)
    return img


def resample_seg_to_pt(seg_img, pt_img):
    """SEG -> PT grid via NN (P80 D4 finding: SEG and PT come on different grids)."""
    rs = sitk.ResampleImageFilter()
    rs.SetReferenceImage(pt_img)
    rs.SetInterpolator(sitk.sitkNearestNeighbor)
    rs.SetDefaultPixelValue(0)
    return rs.Execute(seg_img)


def process_patient(zf, pid, ehr_row):
    pt_member  = f'{DATA_ROOT}{pid}/{pid}__PT.nii.gz'
    seg_member = f'{DATA_ROOT}{pid}/{pid}.nii.gz'
    pt_img = read_nifti_from_zip(zf, pt_member)
    seg_img = read_nifti_from_zip(zf, seg_member)
    seg_on_pt = resample_seg_to_pt(seg_img, pt_img)

    suv = sitk.GetArrayFromImage(pt_img).astype(np.float32)        # (z, y, x)
    seg = sitk.GetArrayFromImage(seg_on_pt).astype(np.uint8)
    spacing_xyz = pt_img.GetSpacing()                               # (sx, sy, sz)
    voxel_spacing_zyx = (float(spacing_xyz[2]), float(spacing_xyz[1]), float(spacing_xyz[0]))

    centre_id = ehr_row.get('CenterID')
    centre_name, vendor = lookup_centre(centre_id)

    patient_meta = {
        'task1_patient': bool(ehr_row.get('Task 1') == 1),
        'task2_patient': bool(ehr_row.get('Task 2') == 1),
        'Relapse':  ehr_row.get('Relapse'),
        'RFS':      ehr_row.get('RFS'),
        'T-stage':  ehr_row.get('T-stage'),
        'N-stage':  ehr_row.get('N-stage'),
        'M-stage':  ehr_row.get('M-stage'),
        'HPV Status': ehr_row.get('HPV Status'),
    }

    rows = extract_hecktor_lesions(
        suv=suv, seg_multi=seg, voxel_spacing_mm=voxel_spacing_zyx,
        case_id=pid, centre_id=int(centre_id) if pd.notna(centre_id) else None,
        centre_name=centre_name, vendor=vendor, patient_meta=patient_meta,
    )
    return rows


N_SMOKE = 3
print(f'Smoke test: {N_SMOKE} patients')

# ===== Cell 7: Production extraction (~30-60 min) =====
import time
import json
import pandas as pd
import zipfile

OUT_DIR = '/content/drive/MyDrive/P79 Data/hecktor'
import os
os.makedirs(OUT_DIR, exist_ok=True)

OUT_PARQUET = f'{OUT_DIR}/hecktor_lesions.parquet'
OUT_LOG_CSV = f'{OUT_DIR}/hecktor_extraction_log.csv'

# Resume support: if parquet exists, skip patients already in it
done_pids = set()
if os.path.exists(OUT_PARQUET):
    prev = pd.read_parquet(OUT_PARQUET, columns=['case_id'])
    done_pids = set(prev['case_id'].unique())
    print(f'Resume: {len(done_pids)} patients already in {OUT_PARQUET}; will skip')

t0 = time.time()
all_rows = []
log_records = []
n_done = 0
n_skipped = 0
n_failed = 0

with zipfile.ZipFile(ZIP_PATH) as zf:
    for i, (_, r) in enumerate(cohort.iterrows(), 1):
        pid = r['PatientID']
        if pid in done_pids:
            n_skipped += 1
            continue
        try:
            t_start = time.time()
            rows = process_patient(zf, pid, r)
            dt = time.time() - t_start
            all_rows.extend([row.__dict__ for row in rows])
            log_records.append({
                'patient_id': pid, 'n_lesions': len(rows),
                'elapsed_sec': dt, 'status': 'ok',
                'centre_id': r.get('CenterID'),
            })
            n_done += 1
            if n_done % 25 == 0:
                rate = n_done / (time.time() - t0)
                eta_min = (len(cohort) - n_done - n_skipped) / max(rate, 1e-6) / 60.0
                print(f'  [{i}/{len(cohort)}] {pid}: {len(rows)} lesions, '
                      f'rate {rate:.2f} patients/s, ETA {eta_min:.1f} min')
        except Exception as e:
            n_failed += 1
            log_records.append({
                'patient_id': pid, 'n_lesions': 0, 'elapsed_sec': 0,
                'status': f'FAILED: {e}', 'centre_id': r.get('CenterID'),
            })
            print(f'  [{i}/{len(cohort)}] {pid}: FAILED: {e}')

# Append to existing parquet (if any) and write
if all_rows:
    new_df = pd.DataFrame(all_rows)
    if os.path.exists(OUT_PARQUET):
        prev = pd.read_parquet(OUT_PARQUET)
        out = pd.concat([prev, new_df], ignore_index=True)
    else:
        out = new_df
    out.to_parquet(OUT_PARQUET, index=False)
    print(f'\nWrote {OUT_PARQUET}: {len(out)} rows total')

pd.DataFrame(log_records).to_csv(OUT_LOG_CSV, index=False, mode='a',
                                  header=not os.path.exists(OUT_LOG_CSV))

elapsed = time.time() - t0
print(f'\nDone: {n_done} new, {n_skipped} resumed/skipped, {n_failed} failed in {elapsed/60:.1f} min')

import hashlib
import pandas as pd

def _sha256(p):
    h = hashlib.sha256()
    with open(p, 'rb') as f:
        for chunk in iter(lambda: f.read(1<<20), b''):
            h.update(chunk)
    return h.hexdigest()

                h.update(chunk)
        print(f'  staged {name}: {sz:,} bytes, SHA-256 {h.hexdigest()}')
