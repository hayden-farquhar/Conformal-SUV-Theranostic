# ===== Cell 5 (CONDITIONAL): Regenerate SUV NIfTIs for AutoPET-III subsample =====
# Only run if Cell 4 reported needs_suv_regen > 0.
# Per series: (i) unzip DICOM if zipped-but-not-extracted, (ii) extract per-series
# PETMetadata from any DICOM slice, (iii) regenerate paired SUV NIfTI from DICOM
# via the §3.3-validated suv_conversion pipeline.

import os, zipfile, glob
import pandas as pd
from pathlib import Path
import SimpleITK as sitk
from src.preprocess.suv_conversion import dicom_series_to_suv_sitk, extract_pet_metadata

PRE_CSV = f'{OUT_DIR}/phase2_preflight_autopet_iii.csv'
SUV_OUT_DIR = Path(f'{DATA_ROOT}/autopet_iii/paired_inputs')
SUV_OUT_DIR.mkdir(parents=True, exist_ok=True)

pre = pd.read_csv(PRE_CSV)
todo = pre[pre['needs_suv_regen']]
print(f'{len(todo)} series need SUV regen')
print()

n_unzipped = 0
n_regen = 0
n_failed = 0

for i, (_, r) in enumerate(todo.iterrows(), 1):
    series_uid = r['series_uid']
    dicom_dir = Path(r['dicom_dir'])
    dicom_zip = Path(r['dicom_zip'])
    out = SUV_OUT_DIR / f'{series_uid}_0001.nii.gz'

    if out.exists():
        print(f'[{i}/{len(todo)}] {series_uid[:40]}... -- SUV already exists, skipping')
        continue

    # Step 1: ensure DICOM is extracted (unzip if needed)
    if not dicom_dir.exists():
        if dicom_zip.exists():
            print(f'[{i}/{len(todo)}] {series_uid[:40]}... -- unzipping {dicom_zip.stat().st_size/1024/1024:.1f} MB')
            try:
                dicom_dir.mkdir(parents=True, exist_ok=True)
                with zipfile.ZipFile(dicom_zip, 'r') as zf:
                    zf.extractall(dicom_dir)
                # Some zips wrap content in a sub-directory; flatten if so
                inside = list(dicom_dir.iterdir())
                if len(inside) == 1 and inside[0].is_dir():
                    nested = inside[0]
                    for f in nested.iterdir():
                        f.rename(dicom_dir / f.name)
                    nested.rmdir()
                n_unzipped += 1
            except Exception as e:
                print(f'    UNZIP FAILED: {e}')
                n_failed += 1
                continue
        else:
            print(f'[{i}/{len(todo)}] {series_uid[:40]}... -- no DICOM dir or zip; skipping')
            n_failed += 1
            continue

    # Step 2: extract PETMetadata + run SUV regen via §3.3-validated pipeline
    try:
        # Find any DICOM slice in the series for metadata extraction
        dcm_candidates = sorted(dicom_dir.glob('*.dcm'))
        if not dcm_candidates:
            # some Drive layouts have no .dcm extension; fall back to any file
            dcm_candidates = sorted(p for p in dicom_dir.iterdir() if p.is_file())
        if not dcm_candidates:
            raise FileNotFoundError(f'no DICOM files in {dicom_dir}')
        meta = extract_pet_metadata(str(dcm_candidates[0]))

        suv_image = dicom_series_to_suv_sitk(str(dicom_dir), meta)
        sitk.WriteImage(suv_image, str(out))
        print(f'[{i}/{len(todo)}] {series_uid[:40]}... -- wrote SUV')
        n_regen += 1
    except Exception as e:
        print(f'[{i}/{len(todo)}] {series_uid[:40]}... -- SUV REGEN FAILED: {e}')
        n_failed += 1

print()
print(f'Done: {n_regen} SUV regenerated, {n_unzipped} DICOMs unzipped, {n_failed} failed.')

    (f'{OUT_PREFIX_SENS}_replicates.parquet',    'phase2_autopet_i_sensitivity_replicates.parquet'),
]

def _sha256(p):
    h = hashlib.sha256()
    with open(p, 'rb') as fh:
        for chunk in iter(lambda: fh.read(1 << 20), b''):
            h.update(chunk)
    return h.hexdigest()

def _stage(items, dst_dir, label):
    print(f'--- {label} ---')
    for src, name in items:
        if not os.path.exists(src):
            print(f'  MISSING: {src}'); continue
        dst = os.path.join(dst_dir, name)
        shutil.copy(src, dst)
        sz = os.path.getsize(dst); sha = _sha256(dst)
        print(f'  {name}  {sz:>12,} bytes  SHA-256 {sha}')

_stage(SMALL, STAGE_SMALL, 'small artefacts (pull back to local for OSF):')
_stage(LARGE, STAGE_LARGE, 'large artefacts (Drive-only; reproducible from wCV + code):')
